Use simple functions to evaluate input values, clear input, and show results:

function appval(val, target) {
  document.getElementById(target).value += val;
}
function clr(target) {
  document.getElementById(target).value = '';
}
function cal(target) {
  try {
    let output = eval(document.getElementById(target).value);
    document.getElementById(target).value = output;
  } catch {
    document.getElementById(target).value = "Error";
  }
}
